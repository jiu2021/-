var search = function(nums, target) {
    if(nums.length == 0) return -1;
    let low = 0;
    let high = nums.length;
      // let high = nums.length - 1; //闭区间
    while(low < high) {
    // while(low <= high) 
        let mid = low + Math.floor((high - low) / 2);
        if(target == nums[mid]) {
            return mid;
        }else if(target > nums[mid]) {
            low = mid + 1 ;
        }else if(target < nums[mid]) {
            high = mid;
              // high = mid - 1;
        }
    }
    return -1;
};

```javascript
// 从头到尾输出链表
var reversePrint = function(head) {
    let res = []
    recur(head,res)
    return res;
};

function recur(head,res){
    let p = head;
    if(p) {
        if(p.next) {
            recur(p.next,res)
        }
        res.push(p.val)
    }
}
```

## dfs

一般解决连通性问题

```javascript
// 岛屿最大面积
var maxAreaOfIsland = function(grid) {
    function dfs(grid,i,j) {
        const r_len = grid.length, c_len = grid[0].length;
          // 先做出界判断 
        if(i<0 || i>=r_len || j<0 || j>=c_len) return 0;

          // 再题目逻辑判断 
        if(grid[i][j] == 0) {
            return 0;
        }else {
              // 搜索过记录
            grid[i][j] = 0;
            return 1+dfs(grid,i-1,j)+dfs(grid,i,j+1)+dfs(grid,i+1,j)+dfs(grid,i,j-1);
        }
    }

    let ans = 0;
    for(let i = 0; i < grid.length; i++) {
        for(let j = 0; j < grid[0].length; j++) {
            ans = Math.max(ans,dfs(grid,i,j));
        }
    }
    return ans
};
```

## bfs

一般解决最短路径、树的层序遍历问题

```javascript
// 腐烂的橘子
var orangesRotting = function(grid) {
    let mat = grid;
    let n = mat.length, m = mat[0].length;

    // 入栈队列
    let queue = [];
    let directions = [[0, 1],[0, -1],[-1, 0],[1, 0]]
    for(let i=0; i<n; i++) {
        for(let j=0; j<m; j++) {
            if(mat[i][j] === 2){
                queue.push([i, j, 0])
            }
        }
    }

    let d = 0;
    while(queue.length) {
        let row = 0,col = 0;
        [row, col, d] = queue.shift();
        // 遍历上下左右的值
        for (let dir of directions) {
            let newI = row + dir[0],
                newJ = col + dir[1];

            if(newI < 0 || newJ < 0 || newI >= n || newJ >= m) continue;

            if( mat[newI][newJ] == 1) {
                mat[newI][newJ] = 2;
                queue.push([newI, newJ, d+1])
            }
        }
    }

    if(mat.flat().indexOf(1) != -1) return -1;
    return d;
};
```

## 回溯

```javascript
// 全排列
var permute = function(nums) {
    const res = [];
    const used = new Array(nums.length).fill(false);
    backtrack(nums,[],res,used);
    return res
};

var backtrack = function(nums,track,res,used) {
        if(nums.length == track.length) {
            res.push([...track]);     //track一直在变，需要拷贝
            return;
        }
        for(let i = 0; i < nums.length; i++) {
            if(!used[i]) {
                used[i] = true;
                track.push(nums[i]);
                backtrack(nums,track,res,used);
                used[i] = false;
                track.pop();
            }
        }
    }
```

// 子集
var subsets = function(nums) {
    let res = [];
    backtrack(0,nums,[],res)
    res.unshift([]);
    return res
};

var backtrack = function(cur,nums,track,res) {
    if(cur > nums.length-1) {
        return;
    }

    for(let i = cur; i < nums.length; i++) {
        track.push(nums[i]);
        res.push([...track])
        backtrack(i+1,nums,track,res);
        track.pop()
    }
}
